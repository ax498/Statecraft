## 〇、核心流程主循环

```java
初始化() {
    注册();  // 各种需要注册的东西：Mod、卡牌、日志等
    ...
}

对决初始化() {
    创建牌堆();
    创建玩家列表();
    渲染();
    状态 = "抽牌";
}

// 同名矛盾检查：回合牌堆中是否存在参与矛盾判定的同名卡牌
矛盾检查() {
    // 实现：遍历回合牌堆，若有任意两张 count_towards_contradiction 且 card_name 相同，返回 true
    return 检测同名矛盾() != null;
}

cheek() {
    if (牌堆.size() == 0) {
        状态 = "游戏结束";
        return true;
    }
    return false;
}

draw() {
    Card card = 牌堆[0];
    牌堆.remove(0);
    回合牌堆.add(card);
    渲染();

    if (矛盾检查()) { 状态 = "矛盾"; return true; }
    card.effect();
    if (矛盾检查()) { 状态 = "矛盾"; return true; }  // 九牛一毛夺取后可能矛盾
    if (cheek()) { return true; }
    return false;
}

主循环() {
    switch (状态) {
        case "初始化":
            初始化();
            break;
        case "进入游戏":
            对决初始化();
            break;
        case "抽牌":
            if (getPlayer().type() != "AI") {
                监听(抽牌按钮松开) {
                    if (draw()) { break; }
                }
                监听(回合结束按钮松开) { 状态 = "回合结束"; break; }
            } else {
                AIDepend();
                break;
            }
        case "矛盾":
            按钮禁用(抽牌按钮);
            按钮禁用(回合结束按钮);
            播放特效(矛盾);
            回合牌堆.clear();
            PlayerNext();
            按钮恢复(抽牌按钮);
            按钮恢复(回合结束按钮);
            状态 = "抽牌";
            渲染();
            break;
        case "回合结束":
            Player player = getPlayer();
            player.cards.add(回合牌堆);
            播放特效(回合结束);  // 卡牌飞向该玩家的手牌堆
            回合牌堆.clear();
            渲染();
            PlayerNext();
            状态 = "抽牌";
            渲染();
            break;
        case "游戏结束":
            if (!弹窗) { 创建弹窗(); }
            监听(重新开始按钮松开) { reset(); 状态 = "进入游戏"; }
            监听(返回按钮松开) { 返回模式选择界面(); }
            break;
    }
}
```

## 二、核心引擎

### 2.1 卡牌管理器 (CardManager)

```java
// scripts/core/card_manager.gd
class 卡牌管理器 {
    // 牌堆与手牌
    List<卡牌资源> 全局牌堆;
    List<卡牌资源> 回合牌堆;      // 本回合抽到的牌
    Dictionary<int, List<卡牌资源>> 玩家手牌;  // 0=玩家, 1=对手, ...
    List<卡牌资源> 弃牌堆;

    void 构建牌堆() {
        全局牌堆.clear();
        卡牌注册表 注册表 = new 卡牌注册表();
        for (卡牌数据 : 注册表.获取全部核心卡牌()) {
            按数据追加卡牌到(卡牌数据, 全局牌堆);
        }
        for (卡牌数据 : Mod注册的卡牌) {
            按数据追加卡牌到(卡牌数据, 全局牌堆);
        }
        洗牌(全局牌堆);
        通知牌堆数量变化(全局牌堆.size());
    }

    卡牌资源 矛盾检查() {
        // 返回触发同名矛盾的牌，无则 null
        return 检测同名矛盾();
    }

    Array 抽牌(int 玩家索引) {
        触发钩子("pre_draw_card", [玩家索引]);
        卡牌资源 卡牌 = 从牌堆顶抽取(全局牌堆);
        触发钩子("post_draw_card", [卡牌, 玩家索引]);

        if (卡牌被Mod消费) {
            发出信号(card_drawn_consumed, 卡牌);
            if (卡牌.强制结束回合) 发出信号(force_end_turn);
            return [索引, 卡牌];
        }

        回合牌堆.add(卡牌);
        卡牌资源 矛盾牌 = 矛盾检查();  // 即 check_name_contradiction

        if (矛盾牌 != null) {
            触发矛盾(矛盾牌, 玩家索引);
            return [索引, 卡牌];
        }

        if (!卡牌.强制结束回合) {
            战斗管理器.请求触发卡牌效果(卡牌, 玩家索引, false);
        }
        发出信号(card_drawn, 卡牌);
        if (卡牌.强制结束回合) 发出信号(force_end_turn);
        return [索引, 卡牌];
    }

    卡牌资源 检测同名矛盾() {
        Map<String, 卡牌资源> 已见名称 = new HashMap<>();
        for (卡牌 : 回合牌堆) {
            if (!卡牌.参与矛盾判定) continue;
            if (已见名称.containsKey(卡牌.卡牌名称))
                return 卡牌;
            已见名称.put(卡牌.卡牌名称, 卡牌);
        }
        return null;
    }

    void 触发矛盾(卡牌资源 触发牌, int 玩家索引) {
        List<卡牌资源> 涉及卡牌 = 回合牌堆.副本();
        List<int> 被保护索引 = 计算自圆其说保护索引();
        // 自圆其说：保护排在它及它之前的卡牌收入手牌
        for (i : 被保护索引) {
            玩家手牌[玩家索引].add(回合牌堆[i]);
        }
        for (i : 倒序遍历(回合牌堆)) {
            if (!被保护索引.contains(i))
                弃牌堆.add(回合牌堆[i]);
        }
        回合牌堆.clear();
        发出信号(contradiction_triggered, 涉及卡牌, 被保护索引, 玩家索引);
        发出信号(force_end_turn);
    }

    void 回合结算(int 玩家索引) {
        if (检测同名矛盾() != null) {
            触发矛盾(...);
            return;
        }
        int 回合总分 = 0;
        for (卡牌 : 回合牌堆) {
            回合总分 += 卡牌.分值;
            玩家手牌[玩家索引].add(卡牌);
        }
        发出信号(turn_settled, 回合总分);
        回合牌堆.clear();
        发出信号(hand_updated, 玩家索引);
    }

    boolean 从手牌转移到回合区(int 玩家索引, 卡牌资源 卡牌) {
        List 手牌 = 玩家手牌[玩家索引];
        int idx = 手牌.indexOf(卡牌);
        if (idx >= 0) {
            手牌.remove(idx);
            回合牌堆.add(卡牌);
            发出信号(hand_updated, 玩家索引);
            return true;
        }
        return false;
    }

    void 重置为新游戏() {
        清空所有手牌();
        回合牌堆.clear();
        弃牌堆.clear();
        构建牌堆();
    }
}
```

### 2.2 战斗管理器 (BattleManager)

```java
// scripts/core/battle_manager.gd
class 战斗管理器 {
    enum 状态 {
        开始战斗, 玩家回合, 处理中, 矛盾, 游戏结束
    }

    void 对决初始化() {
        // 创建牌堆、创建玩家列表、状态设为抽牌
        卡牌管理器.重置为新游戏();
        重置为开始();
    }

    boolean cheek() {
        // 牌堆空则游戏结束，返回 true
        if (牌堆为空()) { 检查游戏结束(); return true; }
        return false;
    }

    状态 当前状态;
    int 当前玩家索引;
    int 玩家数量 = 2;
    Dictionary<int, int> 玩家分数;
    boolean 特效播放中 = false;
    boolean Mod劫持结算 = false;

    void 请求抽牌() {
        if (特效播放中 || 当前状态 != 玩家回合) return;
        if (牌堆为空()) {
            检查游戏结束();
            return;
        }
        if (联机模式 && 非主机) {
            RPC请求主机抽牌();
            return;
        }
        切换到(处理中);
        卡牌管理器.抽牌(当前玩家索引);
    }

    void 请求结束回合() {
        if (特效播放中 || 当前状态 != 玩家回合) return;
        if (回合牌堆为空()) {
            下一回合();
            return;
        }
        if (联机模式 && 非主机) {
            RPC请求主机结束回合();
            return;
        }
        切换到(处理中);
        卡牌管理器.回合结算(当前玩家索引);
        等待收牌动画完成后切换回合();
    }

    void 抽牌动画完成() {
        if (Mod劫持结算) return;  // 九牛一毛、出奇制胜等选择中
        if (特效播放中) return;
        if (联机 && 客户端) {
            // 客户端收到主机通知后处理九牛一毛/出奇制胜/偷梁换柱选择
            处理待定选择();
            return;
        }
        触发钩子("post_draw_animation", [当前玩家索引]);
        if (九牛一毛待定) 处理九牛一毛选择();
        else if (出奇制胜待定) 处理出奇制胜选择();
        else if (偷梁换柱待定) 处理偷梁换柱选择();
        else if (Mod劫持结算) return;
        else {
            切换到(玩家回合);
            if (当前是AI) 启动AI回合();
        }
    }

    void 下一回合() {
        int 新索引 = (当前玩家索引 + 1) % 玩家数量;
        if (联机 && 主机) {
            RPC同步回合(新索引);
        } else {
            执行回合切换(新索引);
        }
    }

    void 执行回合切换(int 新索引) {
        当前玩家索引 = 新索引;
        卡牌管理器.回合牌堆.clear();
        切换到(玩家回合);
        if (!联机 || 主机) 检查游戏结束();
    }

    void 检查游戏结束() {
        if (!牌堆为空()) return;
        int 胜者 = 计算最高分玩家();
        boolean 平局 = 存在并列最高();
        if (平局) 胜者 = -1;
        发出信号(game_ended, 结果文本, 玩家0分, 玩家1分);
    }

    void 请求触发卡牌效果(卡牌资源 卡牌, int 玩家索引, boolean 被夺取后) {
        switch (卡牌.技能标识) {
            case "chayanguanse":
                下一张 = 卡牌管理器.窥牌();
                if (下一张 != null)
                    请求显示窥牌(下一张, 是否展示给本地);
                恢复流程不结算();
                break;
            case "jiuniuyimao":
                if (对手有牌) {
                    Mod劫持结算 = true;
                    九牛一毛待定 = true;
                    发出信号(jiuniuyimao_selection_requested, 玩家索引);
                } else 恢复流程不结算();
                break;
            case "chuqizhisheng":
                if (对手有牌) {
                    Mod劫持结算 = true;
                    出奇制胜待定 = true;
                    发出信号(chuqizhisheng_selection_requested, 玩家索引);
                } else 恢复流程不结算();
                break;
            case "stop":
                完成见好就收效果();  // 特效播完后结算
                break;
            case "toulianghuanzhu":
                if (回合区至少2张) {
                    Mod劫持结算 = true;
                    偷梁换柱待定 = true;
                    发出信号(toulianghuanzhu_selection_requested, 玩家索引);
                } else 恢复流程不结算();
                break;
            default:
                触发钩子("trigger_card_effect", [卡牌, 玩家索引]);
        }
    }

    void 应用九牛一毛选择(Dictionary 选择) {
        九牛一毛待定 = false;
        String 牌名 = 选择["card_name"];
        int 对手索引 = 选择["opponent_index"];
        卡牌资源 目标牌 = 从对手手牌找最低分同名牌(对手索引, 牌名);
        if (目标牌 == null) { 恢复流程不结算(); return; }
        发出信号(jiuniuyimao_steal_animation_started, 目标牌, 对手索引, ...);
        卡牌管理器.从手牌转移到回合区(对手索引, 目标牌);
        待定夺取牌效果 = {卡牌: 目标牌, 夺取者: 当前玩家索引};
    }
}
```

```java
// scenes/battle/battle_test.gd
class 战斗测试 extends Control {
    卡牌管理器 卡牌管理器;
    战斗管理器 战斗管理器;
    boolean 抽牌动画进行中 = false;
    boolean 本回合已矛盾 = false;
    List<int> 矛盾被保护索引;

    void 就绪() {
        设置背景();
        连接信号();
        if (论策模式) {
            卡牌管理器.重置为新游戏();
            战斗管理器.重置为开始();
        }
        刷新手牌显示();
        更新分数();
    }

    void 点击抽牌() {
        战斗管理器.请求抽牌();
    }

    void 点击结束回合() {
        战斗管理器.请求结束回合();
    }

    void 当抽到卡牌(卡牌资源 卡牌) {
        if (联机客户端) return;  // 由 client_draw_received 处理
        添加卡牌到回合区显示(卡牌);
        追加日志(格式化卡牌日志(卡牌));
        延迟执行(() -> 播放抽牌动画(卡牌UI, 卡牌));
    }

    void 播放抽牌动画(Control 卡牌UI, 卡牌资源 卡牌) {
        抽牌动画进行中 = true;
        战斗管理器.设置特效播放(true);
        Tween滑入缩放(卡牌UI, () -> {
            播放卡牌专属特效(卡牌UI, 卡牌, () -> {
                抽牌动画进行中 = false;
                战斗管理器.设置特效播放(false);
                战斗管理器.抽牌动画完成();
            });
        });
    }

    void 播放卡牌专属特效(Control 卡牌UI, 卡牌资源 卡牌, Callable 完成回调) {
        switch (卡牌.卡牌名称) {
            case "察言观色": 紫色光晕动画(卡牌UI, 完成回调); break;
            case "九牛一毛": 橙色光晕动画(卡牌UI, 完成回调); break;
            case "见好就收": 金光弹跳动画(卡牌UI, () -> {
                战斗管理器.完成见好就收效果();
                完成回调();
            }); break;
            case "箭在弦上": 蓄力动画(卡牌UI, () -> {
                战斗管理器.请求箭在弦上强制抽牌();
            }); break;
            default: 完成回调();
        }
    }
```

```java
class Mod加载器 {
    List<RefCounted> 已加载Mod实例;

    void 加载Mod脚本() {
        扫描Mod目录下所有gd文件();
        for (路径 : 脚本路径列表) {
            if (ModManager.该Mod已启用(路径对应的ModId)) {
                实例 = 加载并实例化(路径);
                已加载Mod实例.add(实例);
                if (实例.hasMethod("_mod_init"))
                    实例._mod_init();
            }
        }
    }

    void 触发钩子(String 钩子名, Array 参数) {
        for (mod : 已加载Mod实例) {
            if (mod.hasMethod("_on_hook_triggered"))
                mod._on_hook_triggered(钩子名, 参数);
        }
    }

    Variant API调用(String 目标, String 方法名, Array 参数) {
        if (目标 == "CardManager") return 卡牌管理器.callv(方法名, 参数);
        if (目标 == "BattleManager") return 战斗管理器.callv(方法名, 参数);
        return null;
    }
}
```

## 七、核心流程概览

```
【启动】
  ModManager._ready → 加载mod_config、扫描Mod目录
  ModLoader._ready → 加载启用Mod的gd脚本、调用_mod_init
  CardManager._ready → 构建牌堆
  BattleManager._ready → 监听CardManager信号

【主菜单】
  点击开始游戏 → 显示模式选择（论策/演兵）
  论策 → 选人数(2~8) → 设置is_lunce_mode → 切换战斗场景
  演兵 → 创建/加入房间 → Mod校验 → 主机点开始 → 切换战斗场景

【战斗 - 抽牌】
  玩家点抽牌 → BattleManager.request_draw
    → CardManager.draw_card
    → Mod钩子pre_draw_card
    → 从牌堆顶取牌
    → Mod钩子post_draw_card（他山之石可consume_drawn_card消费）
    → 加入turn_container
    → 检测同名矛盾？→ 触发矛盾
    → 触发卡牌效果（察言观色/九牛一毛/出奇制胜/...）
    → emit card_drawn
    → BattleTest播放抽牌动画
    → 动画结束 → draw_animation_finished
    → 处理九牛一毛/出奇制胜/偷梁换柱选择 或 切换PLAYER_TURN

【战斗 - 结束回合】
  玩家点结束回合 → BattleManager.request_end_turn
    → CardManager.settle_turn
    → Mod钩子pre_turn_settled（一石二鸟双倍等）
    → 计算总分、累加玩家分数
    → Mod钩子post_turn_settled
    → emit hand_updated
    → BattleTest播放收牌飞入手牌动画
    → 动画结束 → collection_animation_finished
    → next_turn

【矛盾】
  同名牌 in turn_container → trigger_contradiction
    → 自圆其说保护：保护卡牌入手牌
    → 未保护入弃牌堆
    → emit contradiction_triggered
    → BattleManager切换CONTRADICTION
    → 播放震动+闪红
    → 1.1秒后 contradiction_finished
    → BattleTest播放矛盾淡出
    → contradiction_effect_complete → next_turn
```
